Module store/reducers/generated/userSlice

This file makes and contains a Redux slice/reducer/state/thunks for the UserService from the generated API client

The slice state can be accessed like so: const { data: users, status, errors, messages, extraState } = useAppSelector( ( state: RootState ) => state.users )

The thunks can be dispatched like so:

const dispatch = useAppDispatch()

dispatch(
  deleteUser( {
    path  : { id: id },
    query : { force: false },
  } )
)
  .unwrap()
  .then( response => {
    console.log( "User deleted", response )
  } )
  .catch( ( err: unknown ) => {
    console.log( "Failed to delete user", err )
  } )
